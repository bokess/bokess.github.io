应用场景

        总公司

    需求:公司要让公司开发部的小林执行努力工作这件事

代码实现

定义公司类

function headOffice(name){//定义一个类,这个类代表公司  
    this.name = name;//公司名字  
    this.departments = [];//定义一个存放公司的部门的数组
};
headOffice.prototype.getDepartment = function(){//一个公司肯定知道有什么部门,所以这里定义了一个获取公司部门的方法
    return this.departments;//前面我定义了一个this.departments的数组存放公司的部门,这里把这个数组返回回去,就把公司的部门获取到了.
};
headOffice.prototype.addDepartment = function(name){//我设了一个部门,那我怎么把部门归到公司下呢?所以我这里定义了为公司添加部门的方法
    this.departments.push(name);//把部门推到存放部门的数组里面
    return this;//这里返回this是为了可以执行链式操作
};
headOffice.prototype.removeDepartment = function(name){//公司改革不想要一个部门.所以也要定义一个移除公司部门的方法
    var index = this.departments.indexOf(name);//首先找到要被移除部门,在数组里的下标.(返回大于负1,证明这个公司存在这个部门.否则不存在)
    if(index > -1){//如果下标大于负1证明找了该部门
        this.departments.splice(index,1);//移除这个部门,splice方法第一参数开始下标,第二参数剪切长度.(注意splice会改变原数组);
    }else{//如果index不大于负证明公司里面没有这个部门
        console.warn("没找到"+name+"部门");//这里在浏览器打印一个警告,说没找到这个部门
    };
    return this;//跟前面一样为了可以执行链式操作
};

定义公司部门类

function department(name){//定义一个类,这个类代表公司部门
        this.name = name;//部门名字
        this.persons = [];//存放公司部门的员工
    };
    department.prototype.getPerson = function(){//一个部门里想知道有什么人,所以要设置获取部门员工的方法
        return this.persons;//把部门存放员工的数组返回去,就获取到了部门的所以员工
    };
    department.prototype.addPerson = function(name){//部门业务扩展需要招人,所以要为部门设置添加员工的方法
        this.persons.push(name);//把员工推到存放员工的数组里面
        return this;//为了可以执行链式操作
    };
    department.prototype.removePerson = function(name){//有些员工经常偷懒,迟到,部门不打算继续用这个人了.所以要为部门设置移除员工的方法
        var index = this.persons.indexOf(name);//首先找到这个员工在员工数组里的下标.(返回大于负1,证明这个部门存在这个员工.否则不存在)
        if(index > -1){//如果大于负1,证明这个人存在
            this.persons.splice(index,1);//把这个人炒鱿鱼,splice方法第一参数开始下标,第二参数剪切长度.(注意splice会改变原数组);
        }else{//如果index不大于负1,证明这个人不存在.
            console.warn("没在到"+name+"这个人");//在浏览器打印警告说这个没找到.
        };
        return this;//为了可以执行链式操作
    };

定义员工类

function person(name){//定义一个类,这个类表公司员工或者部门员工
        this.name = name;//员工名字
    };
    person.prototype.hardWroking = function(){//需求是让小林执行努力工作,所以我在这里为员工定义了一个努力工作的方法.
        //如果需求是让一个部门努力工作,那这个方法应该定义在部门类上吗?答案是否定的.例如:公司让财务部努力工作,最终执行努力工作的是财务部的所有员工,所有这个方法还是要定义在员工类上.
        document.write(this.name+" 努力工作...<br/>");
    };
    person.prototype.sleeping = function(){//这是另外定义的让员工休息好,接着加班的方法.
        document.write(this.name+" 努力休息迎接下次加班...<br/>");
    };

var development = new department("开发部门");
//小李,小曾,小张,小刘,小莫,小林都是入职到开发部门
development.addPerson(xLi).addPerson(xZeng).addPerson(xZhang).addPerson(xLiu).addPerson(xMo).addPerson(xLin);

//小庄,小江,小明是入职到财务部门
finance.addPerson(xZhuang).addPerson(xJiang).addPerson(xMing);

var company = new headOffice("总公司");
//公司下面有开发部门(development)和财务部门(finance)
company.addDepartment(development).addDepartment(finance);

实现需求让小林努力工作

这有人可能会说,我直接用xLin.hardWroking().就可以让小林执行努力工作了.这样是可以实现小林努力工作,但是看清楚xLin这个对象是小林自己本身,自己让自己努力工作(神经病),我还不如一直让自己sleeping呢.所以说这样不合适.

for(var i=0;i<company.departments.length;i++){//用for循环遍历了公司部门的数组
    var dep = company.departments[i];//这里得到了公司的部门
    for(var j=0;j<dep.persons.length;j++){//这里用for循环遍历了公司每个部门里面的员工
        if(dep.persons[j].name === '小林'){//如果找到员工的名字是小林的话
            dep.persons[j].hardWroking();//让小林执行努力工作
        };
    };
};

改需求

前面的需求实现了(在页面是打印出来小林努力工作)，现在项目经理说,我想让整个开发部门努力工作。
实现代码:

for(var i=0;i<company.departments.length;i++){//for循环遍历公司的部门
    var dep = company.departments[i];//找到公司的部门
    if(dep.name === '开发部门'){//如果部门的名字是开发部门
        for(var j=0;j<dep.persons.length;j++){//for循环遍历开发部门里面的员工
            dep.persons[j].hardWroking();//让所有员工执行努力工作
        };
    };
};

效果:

我们到上面看看开发部门是不是这几个人,结果证明是正确的.

封装函数

有人说,如果项目经理说又要让整个财务部门努力工作,又要改for循环代码.太麻烦了,能不能封装成一个函数呢?答案是肯定的.
实现代码:

function companySay(name){//定于一个函数只要公司发话,要谁努力工作就谁努力工作,要那个部门努力工作,那个部门就努力工作.
        for(var i=0;i<company.departments.length;i++){//遍历公司的有部门
            if(name ===""||name === undefined||name === "总公司"){//如果函数传空名字,或者不传名字,或者传的是公司名字.则让全公司执行努力工作
                for(var j=0;j<company.departments[i].persons.length;j++){//如果让全公司努力工作,所有部门下面的员工都要执行努力工作
                    company.departments[i].persons[j].hardWroking();
                };
            };
            if(company.departments[i].name === name){//如果函数传部门名称,则让这个部门下的所有员工执行努力工作.
                var persons = company.departments[i].persons;//找到这个部门的所有员工
                for(var j=0;j<persons.length;j++){
                    persons[j].hardWroking();//让员工执行努力工作
                };
            };
            if(company.departments[i].name !== name){//如果函数传进来的是某个人,则第一次遍历的名字一定不相等.需要再遍历部门找到某个人
                var persons = company.departments[i].persons;//找到公司部门
                for(var k =0;k<persons.length;k++){//遍历公司部门
                    if(persons[k].name === name){//找到这人
                        persons[k].hardWroking();//让他执行努力工作
                    };
                };
            };
        };
    };

当然员工类还有一个sleeping的方法,也可以实现,我在这里就不实现了.根据上面的函数把里面的hardWroking改成sleeping即可.

好了,现在函数封装好了,函数传部门则让全部门执行努力工作,传某个人就让某个人努力工作.你正在开香槟庆祝的时候,公司挣了大钱.公司挣大钱也不知道为我们程序猿涨涨工资,就拿钱去开分公司.
公司挣大钱了,开了分公司.这时候需求要变了.项目经理说我想要整个分公司的人都努力工作,你再回去看你的代码,发现要新建一个分公司类,然后还发现之前封装的函数已经不适合现在的需求了,要重新封装.这次需求实现了下次公司发展好,分公司又开分公司. Oh My God 我已经觉得生无可恋了,什么鬼又要改…
有什么方法可以把代码一次封装实现多变的需求吗?答案是肯定的,这方法就是JavaScript组合设计模式.在下篇文章再来介绍组合设计模式到底是何方神圣.

git常用命令(自用笔记仅供参考)

发表于 2016-08-30   |  

设置名字和邮箱


  $git config –global user.name “your name”

  $git config –global user.email “your email”


建立版本库（仓库）


  $mkdir filename (创建空目录)

  $cd filename (进入目录)

  $pwd (显示当前目录)

  $git init (把这个目录变成Git可以管理的仓库)


把文件添加到版本库


  $git add readme.txt (把文件添加到仓库，可以添加多次git add hello.txt)

  $git commit -m “information” (把文件提交到仓库，-m后面是提交说明，commit 可以提交多个文件)

  $git status (查看结果，掌握仓库当前的状态)

  $git diff (是工作区(work dict)和暂存区(stage)的比较)

  $git diff –cached (是暂存区(stage)和分支(master)的比较)


版本回退


  git log (命令显示从最近到最远的提交日志)

  git log –pretty=oneline (每条提交日志以一行显示
Git必须知道当前是哪个版本，HEAD表示当前版本，上个版本HEAD^，上上个HEAD^^)

  git reset –hard HEAD^ (返回上一个版本)

  git reset –hard commit_id (返回某个版本)

  git reflog (查看历史每次提交的日志)
这样即使回到之前的版本，还是可以再回到现在的版本


管理修改


  $git diff HEAD – readme.txt命令可以查看工作区和版本库里面最新版本的区别

  $git checkout – filename (可以丢弃工作区的修改（–前后有空格）)

一：文件自修改后还没有放到暂存区，撤销后就回到和版本库一样的状态

二：文件已添加到暂存区，又做了修改，撤销后就回到添加到暂存区的状态

当git add 提交到暂存区，在commit之前，可以使用git reset HEAD filename（HEAD表示最新版本）可以把暂存区的修改
撤掉，此时工作区内容并没有撤销修改，然后再使用git checkout – filename ，丢弃工作区的修改

删除文件


  $rm filename (把工作区中的文件删除，也可以直接在文件管理器中删除)

  $git rm filename (从版本库中删除（工作区中也删除了）

只是从工作区把文件删除，$git checkout – filename 可以恢复

从版本库中删除，但是没有commit，$git reset HEAD filename 然后$git checkout –filename,可以恢复

从版本库中删除，并且commit,只能版本回退才能恢复

远程仓库


创建SSH.key


  $ssh-keygen -t rsa -C “your email”
(然后可以再用户目录里找到.ssh目录，里面有id_rsa和id_rsa.pub两个文件，前者是私钥，不能泄露出去。
后者是公钥登陆Github，打开Account settings–>SSH Keys—>Add SSH Key，在key文本框里粘贴id_rsa.pub文件内容)

添加远程库


登陆Github，右上角，Create a new repo—创建一个新仓库,创建之后，新仓库有个SSH（HTTPS也行），里面就是新仓库的地址

$git remote add origin address (（添加新仓库SSH/HTTPS） 在本地仓库目录下，与之关联 )

$git push -u origin master (把本地库的所有内容推送到远程库
时间上市把当前分支master推送到远程，由于远程库是空的，当第一次推送master 分支时，加上-u参数，Git不但会把本地master分支内容推送上远程master分支，还会把本地master分支和远程master分支关联，以后推送或者拉取，可以简化命令)

从远程库克隆


  $git clone address（仓库SSH/HTTPS）

分支管理


  $git checkout -b dev (创建并切换 相当于$git branch dev（创建）$git checkout dev（切换），两条命令)

  $ git branch (查看所有分支，带*号的分支表示当前分支)

  $ git merge dev (表示合并指定分支到当前分支)

  $ git branch -d dev (删除分支，合并完成后就可以删除分支了)

  $ git log –graph –pretty=oneline –abbrev-commit (查看分支合并情况)

分支管理策略


Fast forward模式下，删除分支后，会丢掉分支的信息
强制禁用Fast forward模式，Git会在merge是生成一个新的commit，这样从分支历史上可以看出分支信息

  $git merge –no-ff -m “（说明描述）” dev
实际工作中，master分支应该是非常稳定的，仅用来发布新版本，干活都在新分支上，等到版本发布时，再把新分支合并

bug分支


当你在dev分支上工作进行到一半，没法提交，而此时接到要修复一个bug的任务，可以使用Git的stash功能，
把当前的工作现场封存起来，等以后恢复现场后继续工作

  $ git stash (封存现场确定在哪个分支上修复bug，假定在master上修复，就从master上创建临时分支修复完成后，切换到master，并完成合并，最后删除临时分支回到原来的分支工作)

  $git stash list (查看封存的工作现场)

  $git stash apply (恢复现场，此时stash内容并未删除)

  $git stash drop (删除stash内容)

  $git stash pop (恢复现场并且删除stash)

  $git stash apple stash@{0} (指定恢复哪个现场)


$git branch -D ‘name’

(丢弃一个没有被合并过的分支)

多人协作


远程仓库默认名称是origin

  $git remote (查看远程库信息)
  $git remote -v (显示远程库详细信息推送分支：把该分支上的所有本地提交推送给远程库)

  $git push origin master (推送其他分支，例如dev $git push origin dev)

抓取分支

当从远程库clone时，默认只能看到本地master分支，如果要在dev分支上开发，就必须创建远程origin的dev

分支到本地

  $git checkout -b dev origin/dev

当多人向origin/
dev分支同时提交推送时，推送失败，因为共同的修改有冲突，这时候用 $git pull把最新的提交从

origin/dev上转下来，然后本地合并，再推送

如果git pull提示”no tracking information“,说明本地分支和远程分支的链接关系没有创建，用

  $git branch –set-upstream branch-name origin/branch-name
